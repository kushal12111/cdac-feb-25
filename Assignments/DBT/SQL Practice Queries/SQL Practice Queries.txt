1.
SELECT INITCAP(ENAME) AS Capitalized_ENAME
FROM EMP;

2.
SELECT LPAD(ENAME, 30 + LENGTH(ENAME) / 2, '*') AS Center_Justified
FROM EMP;

3.
SELECT INSTR(ENAME, 'un', 1, 2) AS Second_Occurrence
FROM EMP;

4.
SELECT TO_CHAR(TO_NUMBER(SAL), 'SP') AS Spelled_Salary
FROM EMP
WHERE SAL <= 5000000;

5.
SELECT REGEXP_REPLACE(ENAME, '[aeiouAEIOU]', '') AS ENAME_Without_Vowels
FROM EMP;

6.
SELECT 
    HIREDATE, 
    FLOOR(MONTHS_BETWEEN(SYSDATE, HIREDATE) / 12) AS Years,
    MOD(FLOOR(MONTHS_BETWEEN(SYSDATE, HIREDATE)), 12) AS Months,
    TRUNC(SYSDATE - ADD_MONTHS(HIREDATE, FLOOR(MONTHS_BETWEEN(SYSDATE, HIREDATE)))) AS Days,
    TO_CHAR(SYSDATE - HIREDATE, 'HH24') AS Hours,
    TO_CHAR(SYSDATE - HIREDATE, 'MI') AS Minutes,
    TO_CHAR(SYSDATE - HIREDATE, 'SS') AS Seconds
FROM EMP;
